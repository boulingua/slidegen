\begin{frame}{Grammar — Frequency adverbs}
  \begin{block}{Rule}
    Frequency adverbs go \textbf{before} the main verb, but \textbf{after} \textit{to be}.
  \end{block}
  \begin{itemize}
    \item I \textbf{always} tidy my room.
    \item She is \textbf{never} late for school.
  \end{itemize}
\end{frame}
